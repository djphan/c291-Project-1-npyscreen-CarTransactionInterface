-- List condition description, c_id, or both?
SELECT p.name, l.licence_no, p.addr, p.birthday, l.class, c.c_id, 
        c.description, l.expiring_date 
FROM people p, drive_licence l, driving_condition c, restriction r
WHERE p.sin = l.sin AND
      l.licence_no = r.licence_no AND
      r.r_id = c.c_id AND
      l.licence_no = passed_in_licence_# OR
      p.name = passed_in_name

2. 
-- Should the ticket type data be listed? should every field from ticket
-- be listed?
SELECT t.ticket_no, t.violator_no, t.vehicle_no, t.office_no,
       t.vtype, t.vdate, t.place, t.description, tt.fine
FROM people p, drive_licence l, ticket_type tt, ticket t
WHERE p.sin = l.sin AND
      l.licence_no = t.violator_no AND
      t.vtype = tt.vtype AND
      l.licence_no = passed_in_value OR
      p.sin = passed_in_value

SELECT Sales.Total_sales, Sales.Average_price, Tickets.Total_tickets
FROM 
(SELECT COUNT(s.vehicle_id) as Total_sales, AVG(s.price) as Average_price,
s.vehicle_id as Vehicle_id
FROM auto_sale s
GROUP BY s.vehicle_id
) AS Sales
